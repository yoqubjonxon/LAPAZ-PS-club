<script>
(function() {
  var N = 13;
  var seats = [];
  var dailyTotal = 0;
  var stopId = -1;

  // init
  for (var i = 0; i < N; i++) {
    seats.push({
      id: i + 1,
      type: i < 9 ? 'kabina' : 'zal',
      rate: i < 9 ? 30000 : 25000,
      running: false,
      paused: false,
      startTs: 0,
      accSecs: 0
    });
  }

  function getSecs(s) {
    var n = s.accSecs;
    if (s.running && s.startTs) n += Math.floor((Date.now() - s.startTs) / 1000);
    return n;
  }

  function earned(s) { return (getSecs(s) / 3600) * s.rate; }

  function fmt(sec) {
    var h = Math.floor(sec / 3600);
    var m = Math.floor((sec % 3600) / 60);
    var s = sec % 60;
    return p(h) + ':' + p(m) + ':' + p(s);
  }

  function p(n) { return n < 10 ? '0' + n : '' + n; }

  function money(n) {
    return Math.floor(n).toString().replace(/\B(?=(\d{3})+(?!\d))/g, ',');
  }

  function buildGrid() {
    var g = document.getElementById('grid');
    g.innerHTML = '';
    for (var i = 0; i < seats.length; i++) {
      var s = seats[i];
      var div = document.createElement('div');
      div.id = 'card-' + s.id;
      div.className = 'card' + (s.running ? ' active' : s.paused ? ' paused' : '');

      var dotCls = s.running ? 'green' : s.paused ? 'yellow' : '';
      var zalNum = s.type === 'zal' ? (s.id - 9) : 0;
      var typeLabel = s.type === 'kabina' ? 'Kabina' : 'Zal' + zalNum;
      var html = '';

      html += '<div class="card-top">';
      html += '<span class="card-num">JOY #' + s.id + ' <span style="color:#94a3b8;font-size:11px">' + typeLabel + '</span></span>';
      html += '<span class="dot ' + dotCls + '"></span>';
      html += '</div>';

      html += '<div class="timer" id="t' + s.id + '">' + fmt(getSecs(s)) + '</div>';
      html += '<div class="earned" id="e' + s.id + '">' + money(earned(s)) + ' so\'m</div>';

      if (!s.running && !s.paused) {
        html += '<div class="rate-row">';
        html += '<input type="number" id="ri' + s.id + '" value="' + s.rate + '" min="0" step="1000">';
        html += '<select id="rt' + s.id + '">';
        html += '<option value="kabina"' + (s.type === 'kabina' ? ' selected' : '') + '>Kabina</option>';
        html += '<option value="zal"'    + (s.type === 'zal'    ? ' selected' : '') + '>Zal</option>';
        html += '</select></div>';
        html += '<div class="card-btns"><button class="btn-green" data-start="' + s.id + '">▶ Boshlash</button></div>';
      } else {
        html += '<div style="text-align:center;color:#64748b;font-size:11px;margin-bottom:8px">' + money(s.rate) + ' so\'m/soat</div>';
        html += '<div class="card-btns">';
        html += '<button class="btn-red"    data-stop="'  + s.id + '">⏹ To\'xtat</button>';
        html += '<button class="btn-yellow" data-pause="' + s.id + '">' + (s.paused ? '▶ Davom' : '⏸ Pauza') + '</button>';
        html += '</div>';
      }

      div.innerHTML = html;
      g.appendChild(div);
    }
  }

  function tick() {
    var liveSum = 0, activeN = 0;
    for (var i = 0; i < seats.length; i++) {
      var s = seats[i];
      if (s.running || s.paused) liveSum += earned(s);
      if (s.running) activeN++;
      var te = document.getElementById('t' + s.id);
      var ee = document.getElementById('e' + s.id);
      if (te) te.textContent = fmt(getSecs(s));
      if (ee) ee.textContent = money(earned(s)) + ' so\'m';
    }
    document.getElementById('h-active').textContent = activeN;
    document.getElementById('h-live').textContent = money(liveSum) + ' so\'m';
    document.getElementById('h-total').textContent = money(dailyTotal + liveSum) + ' so\'m';
  }

  function getSeat(id) {
    for (var i = 0; i < seats.length; i++) if (seats[i].id === id) return seats[i];
    return null;
  }

  function doStart(id) {
    var s = getSeat(id);
    if (!s || s.running) return;
    var ri = document.getElementById('ri' + id);
    var rt = document.getElementById('rt' + id);
    if (ri) s.rate = parseInt(ri.value) || s.rate;
    if (rt) s.type = rt.value;
    s.running = true; s.paused = false;
    s.startTs = Date.now(); s.accSecs = 0;
    buildGrid();
  }

  function doPause(id) {
    var s = getSeat(id); if (!s) return;
    if (s.running) {
      s.accSecs += Math.floor((Date.now() - s.startTs) / 1000);
      s.running = false; s.paused = true;
    } else if (s.paused) {
      s.startTs = Date.now(); s.running = true; s.paused = false;
    }
    buildGrid();
  }

  function openModal(id) {
    var s = getSeat(id); if (!s) return;
    stopId = id;
    var mTypeLabel = s.type === 'kabina' ? 'Kabina' : 'Zal' + (s.id - 9);
    document.getElementById('m-seat').textContent = 'JOY #' + s.id + ' · ' + mTypeLabel;
    document.getElementById('m-amount').textContent = money(earned(s));
    document.getElementById('m-time').textContent = fmt(getSecs(s));
    document.getElementById('m-rate').textContent = money(s.rate) + ' so\'m/soat';
    document.getElementById('m-type').textContent = mTypeLabel;
    document.getElementById('modal').className = 'open';
  }

  function closeModal() {
    document.getElementById('modal').className = '';
    stopId = -1;
  }

  function doConfirm() {
    var s = getSeat(stopId); if (!s) return;
    dailyTotal += earned(s);
    s.running = false; s.paused = false; s.startTs = 0; s.accSecs = 0;
    closeModal(); buildGrid();
  }

  // Events via delegation
  document.getElementById('grid').addEventListener('click', function(e) {
    var btn = e.target;
    if (!btn || btn.tagName !== 'BUTTON') return;
    var sid;
    if (btn.dataset.start) { sid = parseInt(btn.dataset.start); doStart(sid); }
    else if (btn.dataset.stop)  { sid = parseInt(btn.dataset.stop);  openModal(sid); }
    else if (btn.dataset.pause) { sid = parseInt(btn.dataset.pause); doPause(sid); }
  });

  document.getElementById('btnStartAll').onclick = function() {
    for (var i = 0; i < seats.length; i++) {
      var s = seats[i];
      if (!s.running && !s.paused) { s.running = true; s.startTs = Date.now(); s.accSecs = 0; }
    }
    buildGrid();
  };

  document.getElementById('btnStopAll').onclick = function() {
    if (!confirm('Barcha seanslarn to\'xtatilsinmi?')) return;
    for (var i = 0; i < seats.length; i++) {
      var s = seats[i];
      if (s.running || s.paused) { dailyTotal += earned(s); s.running = false; s.paused = false; s.startTs = 0; s.accSecs = 0; }
    }
    buildGrid();
  };

  document.getElementById('btnKabina').onclick = function() {
    for (var i = 0; i < seats.length; i++) {
      if (!seats[i].running && !seats[i].paused) { seats[i].type = 'kabina'; seats[i].rate = 30000; }
    }
    buildGrid();
  };

  document.getElementById('btnZal').onclick = function() {
    for (var i = 0; i < seats.length; i++) {
      if (!seats[i].running && !seats[i].paused) { seats[i].type = 'zal'; seats[i].rate = 25000; }
    }
    buildGrid();
  };

  document.getElementById('btnConfirm').onclick = doConfirm;
  document.getElementById('btnCancel').onclick  = closeModal;
  document.getElementById('modal').onclick = function(e) { if (e.target === this) closeModal(); };

  buildGrid();
  setInterval(tick, 1000);
})();
</script>
</body>
